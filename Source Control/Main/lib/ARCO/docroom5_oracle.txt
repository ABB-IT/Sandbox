create table DM_SOURCE (
  SOURCEID     NUMBER NOT NULL,
  SOURCE       VARCHAR2(80) NOT NULL,
  LANGCODE     VARCHAR2(5) NULL);

create table DM_TOPIC (
  TOPICID      NUMBER NOT NULL,
  TOPIC        VARCHAR2(80) NOT NULL);

create table DM_LANG (
  LANGID       NUMBER NOT NULL,
  LANGCODE     VARCHAR2(5) NOT NULL,
  LANGENGLISH  VARCHAR2(50) NULL,
  LANGFRENCH   VARCHAR2(50) NULL,
  LANGDUTCH    VARCHAR2(50) NULL,
  LANGICON     VARCHAR2(20) NULL);

create table DM_UDC (
  UDCID        NUMBER NOT NULL,
  UDCCODE      VARCHAR2(100) NOT NULL,
  DESCR_DUTCH  VARCHAR2(1000) NULL,
  DESCR_FRENCH VARCHAR2(1000) NULL);
